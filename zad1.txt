using System;

class Program
{
    static void Main()
    {
        Random random = new Random();
        Console.WriteLine("Symulacja rzutu kostką...");
        
        for (int i = 1; i <= 5; i++)
        {
            int wynik = random.Next(1, 7);
            Console.WriteLine($"Rzut {i}: {wynik}");
        }
        
        Console.WriteLine("Naciśnij dowolny klawisz, aby zakończyć...");
        Console.ReadKey();
    }
}
